# Carousel

An interactive extension of the [`Reel`](../reel) component to cycle through content.

```jsx react-live
() => {
  const items = Array.from({ length: 5 }, (v, i) => i);

  const ExampleContent = styled.div({
    width: '20rem',
    height: '12rem',
    borderRadius: '0.25em',
    background: colors.neutral[30],
  });

  return (
    <Carousel>
      <CarouselControl>
        <CarouselControlCaption>Scroll Left</CarouselControlCaption>
        <IconChevronLeft />
      </CarouselControl>
      <CarouselScroll>
        {items.map(item => (
          <CarouselScrollItem>
            <ExampleContent />
          </CarouselScrollItem>
        ))}
      </CarouselScroll>
      <CarouselControl>
        <CarouselControlCaption>Scroll Right</CarouselControlCaption>
        <IconChevronLeft />
      </CarouselControl>
    </Carousel>
  );
};
```
