      title: 'Bets',

    actives: Math.floor(Math.random() * 5000),

    <LineChart
      title="KPI Overview"

            options={[
              { label: 'Mobile', value: 'm' },
              { label: 'Desktop', value: 'd' },
            ]}
          />
        </Filter>
      }
      hasUpdates={true}
      onUpdate={() => console.log('onUpdate')}
      onShare={() => console.log('onShare')}

    />
  );
};
